def build_site_results(successful_sites, failed_message=None):

    successful = {str(site) for site in successful_sites if site}
    results = []
    for site in active_sites:
        site_code = site["site_code"]
        item = {
            "site_code": site_code,

                "failed_stage": "BR",
                "error_message": failed_message or "FormQuestionAnswers Bronze copy failed or did not write the site success marker."

        results.append(item)
    return results

        # Old C# still applies the common DateFilterEnabled WHERE after this switch.

        block = (
