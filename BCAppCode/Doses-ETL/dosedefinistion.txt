    ))

site_list = [r["SiteCode"] for r in src_df.select("SiteCode").distinct().collect()]
site_string = ",".join([f"'{s}'" for s in site_list])

if site_string:
    delta_tbl.update(
        condition=f"SiteCode IN ({site_string})",
        set={"RowState": "CAST(false AS BOOLEAN)"}
    )

silver_df = spark.table(silver_table)
merge_condition = "t.SiteCode = s.SiteCode AND t.ExId = s.ExId"

update_dict = {c: f"s.{c}" for c in silver_df.columns if c in src_df.columns}
insert_dict = {c: f"s.{c}" for c in silver_df.columns if c in src_df.columns}

(
    delta_tbl.alias("t")
    .merge(src_df.alias("s"), merge_condition)
    .whenMatchedUpdate(
        condition="t.RowChkSum IS NULL OR s.RowChkSum IS NULL OR t.RowChkSum <> s.RowChkSum",
        set=update_dict
    )
    .whenMatchedUpdate(
        condition="t.RowChkSum = s.RowChkSum",
        set={
            "RowState": "CAST(true AS BOOLEAN)",
            "LastModAt": "current_timestamp()"
        }
    )
    .whenNotMatchedInsert(values=insert_dict)
    .execute()
)

final_df = spark.table(silver_table)
print(f"DoseExcuse Silver rows after merge: {final_df.count()}")

    delta_tbl.update(
        condition=f"SiteCode IN ({site_string}) AND DtDate >= '{rowstate_reset_date}'",
        set={"RowState": "CAST(false AS BOOLEAN)"}
    )

silver_df = spark.table(silver_table)
merge_condition = "t.SiteCode = s.SiteCode AND t.DoseId = s.DoseId"

update_dict = {c: f"s.{c}" for c in silver_df.columns if c in src_df.columns}
insert_dict = {c: f"s.{c}" for c in silver_df.columns if c in src_df.columns}

(
    delta_tbl.alias("t")
    .merge(src_df.alias("s"), merge_condition)
    .whenMatchedUpdate(
        condition="t.RowChkSum IS NULL OR s.RowChkSum IS NULL OR t.RowChkSum <> s.RowChkSum",
        set=update_dict
    )
    .whenMatchedUpdate(
        condition="t.RowChkSum = s.RowChkSum",
        set={
            "RowState": "CASE WHEN (t.BlVoid = true AND t.DtVoid = true) OR (t.CltId < 0 AND t.CltId <> -111) THEN CAST(false AS BOOLEAN) ELSE CAST(true AS BOOLEAN) END",
            "LastModAt": "current_timestamp()"
        }
    )
    .whenNotMatchedInsert(values=insert_dict)
    .execute()
)
